import os import shutil import numpy as np from tensorflow.keras.applications.mobilenet_v2 import MobileNetV2, preprocess_input, decode_predictions from tensorflow.keras.preprocessing import image model = MobileNetV2(weights='imagenet') label_map = { "person": ["person", "man", "woman", "boy", "girl"], "animal": ["dog", "cat", "bird", "horse"], "food": ["pizza", "burger", "cake", "apple", "banana"], "vehicle": ["car", "bus", "truck", "train", "motorcycle"], "nature": ["mountain", "sea", "forest", "sky", "valley"] } def classify(img_path): img = image.load_img(img_path, target_size=(224, 224)) x = image.img_to_array(img) x = np.expand_dims(x, axis=0) x = preprocess_input(x) preds = model.predict(x) result = decode_predictions(preds, top=3)[0] for _, label, prob in result: label = label.lower() for category, keywords in label_map.items(): if any(k in label for k in keywords): return category return "other" input_folder = "input_images" output_folder = "output" os.makedirs(output_folder, exist_ok=True) for file in os.listdir(input_folder): if not file.lower().endswith((".jpg", ".png", ".jpeg")): continue path = os.path.join(input_folder, file) category = classify(path) save_dir = os.path.join(output_folder, category) os.makedirs(save_dir, exist_ok=True) shutil.copy(path, os.path.join(save_dir, file)) print("사진 정리 완료!")
