from flask import Flask, render_template, request, redirect import os import shutil app = Flask(__name__) UPLOAD_FOLDER = "uploads" OUTPUT_FOLDER = "output" os.makedirs(UPLOAD_FOLDER, exist_ok=True) os.makedirs(OUTPUT_FOLDER, exist_ok=True) # 간단 AI 분류 (안정 + 수행평가용) def classify(filename): name = filename.lower() if any(x in name for x in ["dog", "cat", "bird"]): return "animal" elif any(x in name for x in ["pizza", "burger", "food", "cake"]): return "food" elif any(x in name for x in ["car", "bus", "train"]): return "vehicle" elif any(x in name for x in ["selfie", "person", "me"]): return "person" else: return "nature" @app.route("/", methods=["GET", "POST"]) def index(): result = None if request.method == "POST": file = request.files["file"] if file: path = os.path.join(UPLOAD_FOLDER, file.filename) file.save(path) category = classify(file.filename) save_dir = os.path.join(OUTPUT_FOLDER, category) os.makedirs(save_dir, exist_ok=True) shutil.copy(path, os.path.join(save_dir, file.filename)) result = category return render_template("index.html", result=result) if __name__ == "__main__": app.run(debug=True)
